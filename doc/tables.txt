CREATE TABLE item (
   id INTEGER PRIMARY KEY   AUTOINCREMENT,
   url           TEXT      NOT NULL UNIQUE,
   desc          TEXT      NOT NULL,
   year          CHAR(4)      NOT NULL,
   price           CHAR(10)      NOT NULL,
   km           CHAR(10)      NOT NULL,
   location          TEXT     NOT NULL,
   phone           CHAR(10)      NOT NULL,
   img           TEXT      NOT NULL

);
